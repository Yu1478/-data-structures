## 1. A-B数对

**分值：** 10 分

**题目描述**  
给出一串单调不下降的数以及一个数字 C，要求计算出所有 A−B=C 的数对的个数（不同位置的数字一样的数对算不同的数对）。

**输入描述**  
输入共两行。

第一行，两个整数 N,C。

第二行，N 个整数，作为要求处理的那串数。

**输出描述**  
一行，表示该串数中包含的满足 A−B=C 的数对的个数。

**输入输出样例**

**输入**

```
4 1
1 1 2 3
```

**输出**

```
3
```

```c
#include <stdio.h>

#define MAXN 200005

long long arr[MAXN]; // 使用 long long 防止数据范围过大

int main() {
    int n;
    long long c;
    if (scanf("%d %lld", &n, &c) != 2) return 0;
    
    for (int i = 0; i < n; i++) {
        scanf("%lld", &arr[i]);
    }
    
    long long ans = 0;
    int left = 0, right = 0;
    
    while (left < n && right < n) {
        long long diff = arr[right] - arr[left];
        
        if (diff < c) {
            // 差值太小，右指针右移以增大差值
            right++;
        } else if (diff > c) {
            // 差值太大，左指针右移以减小差值
            left++;
        } else {
            // 差值等于 c，统计连续相同的 B 和 A 的个数
            if (left == right) {
                // 如果 c == 0，左右指针会重合，此时直接右移即可
                right++;
                continue;
            }
            
            long long valB = arr[left];
            long long valA = arr[right];
            
            int countB = 0, countA = 0;
            
            // 统计左侧连续等于 valB 的个数
            while (left < n && arr[left] == valB) {
                countB++;
                left++;
            }
            
            // 统计右侧连续等于 valA 的个数
            while (right < n && arr[right] == valA) {
                countA++;
                right++;
            }
            
            // 组合数相乘并累加到答案中
            ans += (long long)countB * countA;
        }
    }
    
    printf("%lld\n", ans);
    
    return 0;
}
```

## 2. 普通二叉树

**分值：** 10 分

**题目描述**  
您需要写一种数据结构，来维护一些数（都是 1e9 以内的数字）的集合，最开始时集合是空的。其中需要提供以下操作，操作次数 q 不超过 10^4：

1. 查询值为 x 的数的排名（排名定义为比当前数小的数的个数 +1。若有多个相同的数，应输出最小的排名）。
2. 查询排名为 x 的数。
3. 求 x 的前驱（前驱定义为小于 x，且最大的数）。若未找到则输出 −2147483647。
4. 求 x 的后继（后继定义为大于 x，且最小的数）。若未找到则输出 2147483647。
5. 插入一个数 x。

**输入描述**  
第一行是一个整数 q，表示操作次数。

接下来 q 行，每行两个整数 op,x，分别表示操作序号以及操作的参数 x。

**输出描述**  
输出有若干行。对于操作 1,2,3,4 输出一个整数，表示该操作的结果。

**输入输出样例**

**输入**

```
7
5 1
5 3
5 5
1 3
2 2
3 3
4 3
```

**输出**

```
2
3
1
5
```

```c
#include <stdio.h>
#include <stdlib.h>

// 定义二叉树节点结构
typedef struct Node {
    int val;
    int count;      // 记录相同值的个数，用于处理重复元素
    struct Node *left, *right;
} Node;

// 创建新节点
Node* newNode(int val) {
    Node* node = (Node*)malloc(sizeof(Node));
    node->val = val;
    node->count = 1;
    node->left = node->right = NULL;
    return node;
}

// 辅助函数：统计以root为根的树的总节点数（包含重复的count）
int countNodes(Node* root) {
    if (root == NULL) return 0;
    return root->count + countNodes(root->left) + countNodes(root->right);
}

// 操作5：插入一个数 x
void insert(Node** root, int val) {
    if (*root == NULL) {
        *root = newNode(val);
        return;
    }
    Node* cur = *root;
    while (1) {
        if (cur->val == val) {
            cur->count++;
            return;
        } else if (val < cur->val) {
            if (cur->left == NULL) {
                cur->left = newNode(val);
                return;
            }
            cur = cur->left;
        } else {
            if (cur->right == NULL) {
                cur->right = newNode(val);
                return;
            }
            cur = cur->right;
        }
    }
}

// 操作1：查询值为x的数的排名（比x小的数的个数 + 1）
int getRank(Node* root, int val) {
    if (root == NULL) return 1;
    if (val == root->val) {
        return countNodes(root->left) + 1;
    } else if (val < root->val) {
        return getRank(root->left, val);
    } else {
        return countNodes(root->left) + root->count + getRank(root->right, val);
    }
}

// 操作2：查询排名为x的数
int getByRank(Node* root, int rank) {
    int leftCount = countNodes(root->left);
    if (rank <= leftCount) {
        return getByRank(root->left, rank);
    } else if (rank <= leftCount + root->count) {
        return root->val;
    } else {
        return getByRank(root->right, rank - leftCount - root->count);
    }
}

// 操作3：求x的前驱（小于x的最大数）
int getPre(Node* root, int val) {
    if (root == NULL) return -2147483647;
    if (root->val >= val) {
        // 当前节点 >= x，前驱一定在左子树
        return getPre(root->left, val);
    } else {
        // 当前节点 < x，当前节点可能是前驱，但也可能在右子树有更大的前驱
        int rightPre = getPre(root->right, val);
        if (rightPre == -2147483647) return root->val;
        return rightPre;
    }
}

// 操作4：求x的后继（大于x的最小数）
int getSuc(Node* root, int val) {
    if (root == NULL) return 2147483647;
    if (root->val <= val) {
        // 当前节点 <= x，后继一定在右子树
        return getSuc(root->right, val);
    } else {
        // 当前节点 > x，当前节点可能是后继，也可能在左子树有更小的后继
        int leftSuc = getSuc(root->left, val);
        if (leftSuc == 2147483647) return root->val;
        return leftSuc;
    }
}

int main() {
    int q;
    if (scanf("%d", &q) != 1) return 0;
    
    Node* root = NULL;
    
    while (q--) {
        int op, x;
        scanf("%d %d", &op, &x);
        
        if (op == 5) {
            insert(&root, x);
        } else if (op == 1) {
            printf("%d\n", getRank(root, x));
        } else if (op == 2) {
            printf("%d\n", getByRank(root, x));
        } else if (op == 3) {
            printf("%d\n", getPre(root, x));
        } else if (op == 4) {
            printf("%d\n", getSuc(root, x));
        }
    }
    
    return 0;
}
```

## 3. 队列安排

**分值：** 10 分

**题目描述**  
一个学校里老师要将班上 N 个同学排成一列，同学被编号为 1∼N，他采取如下的方法：

先将 1 号同学安排进队列，这时队列中只有他一个人；

2∼N 号同学依次入列，编号为 i 的同学入列方式为：老师指定编号为 i 的同学站在编号为 1∼(i−1) 中某位同学（即之前已经入列的同学）的左边或右边；

最后，从队列中去掉 M(M<N）个同学，其他同学位置顺序不变。

在所有同学按照上述方法队列排列完毕后，老师想知道从左到右所有同学的编号。

**输入描述**  
第 1 行为一个正整数 N，表示了有 N 个整数。  
第 2∼N 行，第 i 行包含两个整数 k,p，其中 k 为小于 i 的正整数，p 为 0 或者 1。若 p 为 0，则表示将 i 号同学插入到 k 号同学的左边，p 为 1 则表示插入到右边。  
第 N+1 行为一个正整数 M，表示去掉的同学数目。  
接下来 M 行，每行一个正整数 x，表示将 x 号同学从队列中移去，如果 x 号同学已经不在队列中则忽略这一条指令。

**输出描述**  
1 行，包含最多 N 个空格隔开的正整数，表示了队列从左到右所有同学的编号，行末换行且无空格。

**输入输出样例**

**输入**

```
4
1 0
2 1
1 0
2
3
3
```

**输出**

```
2 4 1
```

```c
#include <stdio.h>

#define MAXN 100005

int left_node[MAXN];  // 记录左侧同学编号
int right_node[MAXN]; // 记录右侧同学编号
int removed[MAXN];    // 标记是否被移除

int main() {
    int N, M;
    if (scanf("%d", &N) != 1) return 0;
    
    // 初始化1号同学，他此时在队列中，左右都没有人
    left_node[1] = 0;
    right_node[1] = 0;
    
    // 处理 2~N 号同学的插入
    for (int i = 2; i <= N; i++) {
        int k, p;
        scanf("%d %d", &k, &p);
        
        if (p == 0) {
            // 插入到 k 的左边
            int k_left = left_node[k];
            
            // 更新 i 的左右指针
            left_node[i] = k_left;
            right_node[i] = k;
            
            // 更新 k 的左指针
            left_node[k] = i;
            
            // 更新 k 原来的左邻居的右指针（如果存在的话）
            if (k_left != 0) {
                right_node[k_left] = i;
            }
        } else {
            // 插入到 k 的右边
            int k_right = right_node[k];
            
            // 更新 i 的左右指针
            left_node[i] = k;
            right_node[i] = k_right;
            
            // 更新 k 的右指针
            right_node[k] = i;
            
            // 更新 k 原来的右邻居的左指针（如果存在的话）
            if (k_right != 0) {
                left_node[k_right] = i;
            }
        }
    }
    
    // 处理移除操作
    scanf("%d", &M);
    for (int i = 0; i < M; i++) {
        int x;
        scanf("%d", &x);
        
        if (removed[x]) continue; // 如果已经移除，忽略
        
        int x_left = left_node[x];
        int x_right = right_node[x];
        
        // 将 x 的左右邻居互相连接
        if (x_left != 0) {
            right_node[x_left] = x_right;
        }
        if (x_right != 0) {
            left_node[x_right] = x_left;
        }
        
        removed[x] = 1; // 标记为已移除
    }
    
    // 寻找队列的最左端（即 left 为 0 且未被移除的节点）
    int head = 1;
    while (left_node[head] != 0) {
        head = left_node[head];
    }
    
    // 从左到右遍历输出
    int cur = head;
    int first = 1; // 用于控制行末无空格
    while (cur != 0) {
        if (!removed[cur]) {
            if (!first) printf(" ");
            printf("%d", cur);
            first = 0;
        }
        cur = right_node[cur];
    }
    printf("\n");
    
    return 0;
}
```

## 4. 明明的随机数

**分值：** 10 分

**题目描述**  
明明想在学校中请一些同学一起做一项问卷调查，为了实验的客观性，他先用计算机生成了 N 个 1 到 1000 之间的随机整数（N≤100），对于其中重复的数字，只保留一个，把其余相同的数去掉，不同的数对应着不同的学生的学号。然后再把这些数从小到大排序，按照排好的顺序去找同学做调查。请你协助明明完成"去重"与"排序"的工作。

**输入描述**  
输入有两行，第 1 行为 1 个正整数，表示所生成的随机数的个数 N。  
第 2 行有 N 个用空格隔开的正整数，为所产生的随机数。

**输出描述**  
输出也是两行，第 1 行为 1 个正整数 M，表示不相同的随机数的个数。  
第 2 行为 M 个用空格隔开的正整数，为从小到大排好序的不相同的随机数。

**输入输出样例**

**输入**

```
10
20 40 32 67 40 20 89 300 400 15
```

**输出**

```
8
15 20 32 40 67 89 300 400
```

```c
#include <stdio.h>

#define MAX_VAL 1005

int main() {
    int n;
    if (scanf("%d", &n) != 1) return 0;
    
    int vis[MAX_VAL] = {0}; // 标记数组，初始化为0，表示数字未出现
    int count = 0;          // 记录不重复数字的个数
    
    for (int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);
        if (vis[x] == 0) {  // 如果这个数字之前没出现过
            vis[x] = 1;     // 标记为出现过
            count++;        // 不重复数字个数加1
        }
    }
    
    // 输出不重复的数字个数
    printf("%d\n", count);
    
    // 从小到大遍历，输出出现过的数字
    int first = 1; // 用于控制行末无空格
    for (int i = 1; i <= 1000; i++) {
        if (vis[i] == 1) {
            if (!first) printf(" ");
            printf("%d", i);
            first = 0;
        }
    }
    printf("\n");
    
    return 0;
}
```

## 5. 奖学金

**分值：** 10 分

**题目描述**  
某小学最近得到了一笔赞助，打算拿出其中一部分为学习成绩优秀的前 5 名学生发奖学金。期末，每个学生都有 3 门课的成绩：语文、数学、英语。先按总分从高到低排序，如果两个同学总分相同，再按语文成绩从高到低排序，如果两个同学总分和语文成绩都相同，那么规定学号小的同学排在前面，这样，每个学生的排序是唯一确定的。

任务：先根据输入的 3 门课的成绩计算总分，然后按上述规则排序，最后按排名顺序输出前五名学生的学号和总分。注意，在前 5 名同学中，每个人的奖学金都不相同，因此，你必须严格按上述规则排序。例如，在某个正确答案中，如果前两行的输出数据（每行输出两个数：学号、总分）是：

```
7 279
5 279
```

这两行数据的含义是：总分最高的两个同学的学号依次是 7 号、5 号。这两名同学的总分都是 279（总分等于输入的语文、数学、英语三科成绩之和），但学号为 7 的学生语文成绩更高一些。如果你的前两名的输出数据是：

```
5 279
7 279
```

则按输出错误处理，不能得分。

**输入描述**  
共 n+1 行。  
第 1 行为一个正整数 n（n≤300），表示该校参加评选的学生人数。  
第 2 到 n+1 行，每行有 3 个用空格隔开的数字，每个数字都在 0 到 100 之间。第 j 行的 3 个数字依次表示学号为 j-1 的学生的语文、数学、英语的成绩。每个学生的学号按照输入顺序编号为 1∼n（恰好是输入数据的行号减 1）。

**输出描述**  
共 5 行，每行是两个用空格隔开的正整数，依次表示前 5 名学生的学号和总分。

**输入输出样例**

**样例 1**

**输入**

```
6
90 67 80
87 66 91
78 89 91
88 99 77
67 89 64
78 89 98
```

**输出**

```
6 265
4 264
3 258
2 244
1 237
```

**样例 2**

**输入**

```
8
80 89 89
88 98 78
90 67 80
87 66 91
78 89 91
88 99 77
67 89 64
78 89 98
```

**输出**

```
8 265
2 264
6 264
1 258
5 258
```

```c
#include <stdio.h>
#include <stdlib.h>

#define MAXN 305

// 定义学生结构体
typedef struct {
    int id;       // 学号
    int chinese;  // 语文成绩
    int total;    // 总分
} Student;

// 自定义比较函数，用于 qsort
int cmp(const void *a, const void *b) {
    Student *s1 = (Student *)a;
    Student *s2 = (Student *)b;
    
    // 1. 总分从高到低
    if (s1->total != s2->total) {
        return s2->total - s1->total;
    }
    // 2. 语文成绩从高到低
    if (s1->chinese != s2->chinese) {
        return s2->chinese - s1->chinese;
    }
    // 3. 学号从小到大
    return s1->id - s2->id;
}

int main() {
    int n;
    if (scanf("%d", &n) != 1) return 0;
    
    Student stu[MAXN];
    
    for (int i = 0; i < n; i++) {
        int math, english;
        scanf("%d %d %d", &stu[i].chinese, &math, &english);
        stu[i].id = i + 1; // 学号从 1 开始，等于输入行号
        stu[i].total = stu[i].chinese + math + english;
    }
    
    // 使用快速排序对前 n 个学生进行排序
    qsort(stu, n, sizeof(Student), cmp);
    
    // 输出前 5 名的学号和总分
    for (int i = 0; i < 5; i++) {
        printf("%d %d\n", stu[i].id, stu[i].total);
    }
    
    return 0;
}
```

## 6. 纪念品分组

**分值：** 10 分

**题目描述**  
元旦快到了，校学生会让乐乐负责新年晚会的纪念品发放工作。为使得参加晚会的同学所获得的纪念品价值相对均衡，他要把购来的纪念品根据价格进行分组，但每组最多只能包括两件纪念品，并且每组纪念品的价格之和不能超过一个给定的整数。为了保证在尽量短的时间内发完所有纪念品，乐乐希望分组的数目最少。

你的任务是写一个程序，找出所有分组方案中分组数最少的一种，输出最少的分组数目。

**输入描述**  
共 n+2 行：  
第一行包括一个整数 w，为每组纪念品价格之和的上限。  
第二行为一个整数 n，表示购来的纪念品的总件数 G。  
第 3∼n+2 行每行包含一个正整数 Pi 表示所对应纪念品的价格。

**输出描述**  
一个整数，即最少的分组数目。

**输入输出样例**

**输入**

```
100 
9 
90 
20 
20 
30 
50 
60 
70 
80 
90
```

**输出**

```
6
```

```c
#include <stdio.h>
#include <stdlib.h>

#define MAXN 30005

int price[MAXN];

// qsort 的比较函数，用于升序排序
int cmp(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int main() {
    int w, n;
    if (scanf("%d %d", &w, &n) != 2) return 0;
    
    for (int i = 0; i < n; i++) {
        scanf("%d", &price[i]);
    }
    
    // 对价格进行升序排序
    qsort(price, n, sizeof(int), cmp);
    
    int left = 0;       // 指向当前最便宜的纪念品
    int right = n - 1;  // 指向当前最贵的纪念品
    int count = 0;      // 记录分组数
    
    while (left <= right) {
        // 最贵的纪念品必须被分走，所以 right 左移，组数加 1
        right--;
        count++;
        
        // 如果最贵的能和最便宜的凑成一组，则最便宜的也被分走，left 右移
        if (left <= right && price[left] + price[right + 1] <= w) {
            left++;
        }
    }
    
    printf("%d\n", count);
    
    return 0;
}
```

## 7. 统计数字

**分值：** 10 分

**题目描述**  
某次科研调查时得到了 n 个自然数，每个数均不超过 1500000000 (1.5×10^9)。已知不相同的数不超过 10000 个，现在需要统计这些自然数各自出现的次数，并按照自然数从小到大的顺序输出统计结果。

**输入描述**  
共 n+1 行。

第一行是整数 n，表示自然数的个数；

第 2 至 n+1 每行一个自然数。

**输出描述**  
共 m 行（m 为 n 个自然数中不相同数的个数），按照自然数从小到大的顺序输出。

每行输出 2 个整数，分别是自然数和该数出现的次数，其间用一个空格隔开。

**输入输出样例**

**输入**

```
8
2
4
2
4
5
100
2
100
```

**输出**

```
2 3
4 2
5 1
100 2
```

```c
#include <stdio.h>
#include <stdlib.h>

#define MAXN 200005 // 假设 n 的最大规模在 20万 左右

long long arr[MAXN]; // 使用 long long 防止数据过大，虽然题目说是自然数，用 long long 最安全

// qsort 的比较函数，用于升序排序
int cmp(const void *a, const void *b) {
    long long x = *(long long *)a;
    long long y = *(long long *)b;
    if (x < y) return -1;
    if (x > y) return 1;
    return 0;
}

int main() {
    int n;
    if (scanf("%d", &n) != 1) return 0;
    
    for (int i = 0; i < n; i++) {
        scanf("%lld", &arr[i]);
    }
    
    // 对所有数字进行升序排序
    qsort(arr, n, sizeof(long long), cmp);
    
    // 遍历排序后的数组，统计并输出
    int count = 1; // 记录当前数字出现的次数，至少为1
    for (int i = 0; i < n; i++) {
        // 如果下一个数字和当前数字相同，计数加1
        if (i + 1 < n && arr[i] == arr[i + 1]) {
            count++;
        } else {
            // 如果下一个数字不同，或者已经是最后一个数字，输出结果
            printf("%lld %d\n", arr[i], count);
            count = 1; // 重置计数器，准备统计下一个新数字
        }
    }
    
    return 0;
}
```

## 8. 分数线划定

**分值：** 10 分

**题目描述**  
世博会志愿者的选拔工作正在 A 市如火如荼的进行。为了选拔最合适的人才，A 市对所有报名的选手进行了笔试，笔试分数达到面试分数线的选手方可进入面试。面试分数线根据计划录取人数的 150% 划定，即如果计划录取 m 名志愿者，则面试分数线为排名第 m×150%（向下取整）名的选手的分数，而最终进入面试的选手为笔试成绩不低于面试分数线的所有选手。

现在就请你编写程序划定面试分数线，并输出所有进入面试的选手的报名号和笔试成绩。

**输入描述**  
第一行，两个整数 n,m（5≤n≤5000,3≤m≤n），中间用一个空格隔开，其中 n 表示报名参加笔试的选手总数，m 表示计划录取的志愿者人数。输入数据保证 m×150% 向下取整后小于等于 n。  
第二行到第 n+1 行，每行包括两个整数，中间用一个空格隔开，分别是选手的报名号 k（1000≤k≤9999）和该选手的笔试成绩 s（1≤s≤100）。数据保证选手的报名号各不相同。

**输出描述**  
第一行，有 2 个整数，用一个空格隔开，第一个整数表示面试分数线；第二个整数为进入面试的选手的实际人数。

从第二行开始，每行包含 2 个整数，中间用一个空格隔开，分别表示进入面试的选手的报名号和笔试成绩，按照笔试成绩从高到低输出，如果成绩相同，则按报名号由小到大的顺序输出。

**输入输出样例**

**输入**

```
6 3 
1000 90 
3239 88 
2390 95 
7231 84 
1005 95 
1001 88
```

**输出**

```
88 5 
1005 95 
2390 95 
1000 90 
1001 88 
3239 88
```

```c
#include <stdio.h>
#include <stdlib.h>

#define MAXN 5005

// 定义选手结构体
typedef struct {
    int id;    // 报名号
    int score; // 笔试成绩
} Applicant;

// 自定义比较函数，用于 qsort
int cmp(const void *a, const void *b) {
    Applicant *s1 = (Applicant *)a;
    Applicant *s2 = (Applicant *)b;
    
    // 1. 成绩从高到低
    if (s1->score != s2->score) {
        return s2->score - s1->score;
    }
    // 2. 成绩相同，报名号从小到大
    return s1->id - s2->id;
}

int main() {
    int n, m;
    if (scanf("%d %d", &n, &m) != 2) return 0;
    
    Applicant app[MAXN];
    for (int i = 0; i < n; i++) {
        scanf("%d %d", &app[i].id, &app[i].score);
    }
    
    // 对所有选手进行排序
    qsort(app, n, sizeof(Applicant), cmp);
    
    // 计算进入面试的排名界限，并获取分数线
    int k = (int)(m * 1.5); // 向下取整
    int line_score = app[k - 1].score; // 第 k 名的分数即为分数线（数组下标从0开始）
    
    // 统计实际进入面试的人数（成绩 >= 分数线）
    int actual_count = 0;
    for (int i = 0; i < n; i++) {
        if (app[i].score >= line_score) {
            actual_count++;
        } else {
            break; // 因为已经排好序，后面的分数只会更低，直接跳出
        }
    }
    
    // 输出第一行：分数线和实际进入面试的人数
    printf("%d %d\n", line_score, actual_count);
    
    // 输出进入面试的选手信息
    for (int i = 0; i < actual_count; i++) {
        printf("%d %d\n", app[i].id, app[i].score);
    }
    
    return 0;
}
```

## 9. 合并果子

**分值：** 10 分

**题目描述**  
在一个果园里，多多已经将所有的果子打了下来，而且按果子的不同种类分成了不同的堆。多多决定把所有的果子合成一堆。

每一次合并，多多可以把两堆果子合并到一起，消耗的体力等于两堆果子的重量之和。可以看出，所有的果子经过 n−1 次合并之后，就只剩下一堆了。多多在合并果子时总共消耗的体力等于每次合并所耗体力之和。

因为还要花大力气把这些果子搬回家，所以多多在合并果子时要尽可能地节省体力。假定每个果子重量都为 1，并且已知果子的种类数和每种果子的数目，你的任务是设计出合并的次序方案，使多多耗费的体力最少，并输出这个最小的体力耗费值。

例如有 3 种果子，数目依次为 1, 2, 9。可以先将 1, 2 堆合并，新堆数目为 3，耗费体力为 3。接着，将新堆与原先的第三堆合并，又得到新的堆，数目为 12，耗费体力为 12。所以多多总共耗费体力 =3+12=15。可以证明 15 为最小的体力耗费值。

**输入描述**  
共两行。  
第一行是一个整数 n（1≤n≤10000），表示果子的种类数。  
第二行包含 n 个整数，用空格分隔，第 i 个整数 ai（1≤ai≤20000）是第 i 种果子的数目。

**输出描述**  
一个整数，也就是最小的体力耗费值。输入数据保证这个值小于 2^31。

**输入输出样例**

**输入**

```
3 
1 2 9
```

**输出**

```
15
```

```c
#include <stdio.h>

#define MAXN 10005

long long heap[MAXN]; // 使用 long long 防止体力总和溢出 int 范围
int size = 0;

// 向上调整（插入新元素时使用）
void siftUp(int i) {
    while (i > 1 && heap[i] < heap[i / 2]) {
        long long temp = heap[i];
        heap[i] = heap[i / 2];
        heap[i / 2] = temp;
        i /= 2;
    }
}

// 向下调整（删除堆顶元素时使用）
void siftDown(int i) {
    while (2 * i <= size) {
        int minChild = 2 * i;
        if (minChild + 1 <= size && heap[minChild + 1] < heap[minChild]) {
            minChild++;
        }
        if (heap[i] <= heap[minChild]) break;
        long long temp = heap[i];
        heap[i] = heap[minChild];
        heap[minChild] = temp;
        i = minChild;
    }
}

// 向堆中插入元素
void push(long long val) {
    heap[++size] = val;
    siftUp(size);
}

// 取出堆顶最小元素
long long pop() {
    long long minVal = heap[1];
    heap[1] = heap[size--];
    siftDown(1);
    return minVal;
}

int main() {
    int n;
    if (scanf("%d", &n) != 1) return 0;
    
    // 读入数据并建堆
    for (int i = 0; i < n; i++) {
        long long x;
        scanf("%lld", &x);
        push(x);
    }
    
    long long totalCost = 0;
    
    // 每次合并两堆，直到只剩下一堆
    while (size > 1) {
        long long a = pop(); // 取出最小的
        long long b = pop(); // 取出第二小的
        long long sum = a + b;
        totalCost += sum;    // 累加消耗的体力
        push(sum);           // 将合并后的新堆放回
    }
    
    printf("%lld\n", totalCost);
    
    return 0;
}
```
