## 1. 括号匹配

**分值：** 10 分

**题目描述**  
使用计算机进行运算时，需要对表达式中括号是否匹配进行检查。例如：`(1+2)/(21+9)` 中括号匹配，`(3+2(+(1+2))(2/3` 中括号不匹配。使用顺序栈实现对表达式中括号是否相匹配的检查。

**输入描述**  
一个表达式（长度不超过 100）。其中符号中仅包含 `0123456789±*/^()` 等基本符号，且表达式中的数字都是一位的。

**输出描述**  
若匹配输出 `Yes`，否则输出 `No`。

**输入输出样例**

**输入**

```
(1+2)/(21+9)
```

**输出**

```
Yes
```

```c
#include <stdio.h>
#include <string.h>

#define MAXSIZE 100

typedef struct {
    char data[MAXSIZE];
    int top;
} SqStack;

// 初始化
void InitStack(SqStack *s) {
    s->top = -1;
}

// 入栈
void Push(SqStack *s, char x) {
    s->data[++s->top] = x;
}

// 出栈
void Pop(SqStack *s) {
    s->top--;
}

// 判断括号是否匹配
int Compare(char str[]) {
    SqStack s;
    InitStack(&s);

    for (int i = 0; str[i] != '\0'; i++) {
        // 左括号入栈
        if (str[i] == '(') {
            Push(&s, '(');
        }
        // 遇到右括号
        else if (str[i] == ')') {
            // 栈空，说明没有对应左括号
            if (s.top == -1) {
                return 0;
            }
            // 弹出一个左括号
            Pop(&s);
        }
    }

    // 最后栈空才匹配
    return s.top == -1 ? 1 : 0;
}

int main() {
    char str[MAXSIZE];
    scanf("%s", str);

    if (Compare(str))
        printf("Yes");
    else
        printf("No");

    return 0;
}
```

## 2. 回文链表

**分值：** 10 分

**题目描述**  
"回文"指正读反读均相同的字符序列，如 `abcdcba` 和 `abba` 均是回文。使用栈这种数据结构判断给定字符序列是否为回文，要求使用链栈实现。

**输入描述**  
一个字符序列，字符序列长度 ≤ 1000。

**输出描述**  
输出为 `true` / `false`，分别表示输入的字符序列是否为回文。

**输入输出样例**

**输入**

```
abcdcba
```

**输出**

```
true
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    char data;
    struct Node *next;
} Node;

typedef struct {
    Node *top;
} Stack;

void initStack(Stack *s) {
    s->top = NULL;
}

void push(Stack *s, char a) {
    Node *newNode = (Node*)malloc(sizeof(Node));
    newNode->data = a;
    newNode->next = s->top;
    s->top = newNode;
}

void pop(Stack *s, char *a) {
    if (s->top == NULL) return;

    Node *p = s->top;
    *a = p->data;
    s->top = p->next;
    free(p);
}

int charlength(char *a) {
    int i = 0;
    while (a[i] != '\0') i++;
    return i;
}

int ishw(char *a) {
    Stack s;
    initStack(&s);

    int n = charlength(a);
    int i = 0;

    while (i < n / 2) {
        push(&s, a[i]);
        i++;
    }

    if (n % 2 != 0) i++;

    char temp;
    while (i < n) {
        pop(&s, &temp);
        if (a[i] != temp) return 0;
        i++;
    }

    return 1;
}

int main() {
    char a[1000];
    scanf("%s", a);

    if (ishw(a))
        printf("true");
    else
        printf("false");

    return 0;
}
```

## 3. 入栈与出栈

**分值：** 10 分

**题目描述**  
给定一个正整数数列（以 `0` 表示输入结束），从第一个数开始，将每一个数入栈，入栈的同时获得一个分数，即该数的数值乘以入栈后栈的大小。请计算将所有元素入栈后的分数和，然后将栈内元素依次输出，要求使用顺序栈。元素个数 < 100，每个元素 < 100。

**输入描述**  
一行，一个正整数数列，以 `0` 结尾。

**输出描述**  
第一行：一个正整数，表示分数和。  
第二行：将所有元素出栈后的结果，空格分隔。

**输入输出样例**

**输入**

```
1 2 3 0
```

**输出**

```
14
3 2 1
```

```c
#include <stdio.h>

#define MAXSIZE 100

typedef struct {
    int data[MAXSIZE];
    int top;
} Stack;

// 初始化
void initStack(Stack *s) {
    s->top = -1;
}

// 入栈
void Push(Stack *s, int x) {
    if (s->top == MAXSIZE - 1) return;
    s->data[++s->top] = x;
}

// 出栈
void Pop(Stack *s, int *x) {
    if (s->top == -1) return;
    *x = s->data[s->top--];
}

void calculate(int *a, int num) {
    int total = 0;
    Stack s;
    initStack(&s);

    for (int i = 0; i < num; i++) {
        Push(&s, a[i]);
        total += (s.top + 1) * a[i];
    }

    printf("%d\n", total);

    int temp;
    while (s.top != -1) {
        Pop(&s, &temp);
        printf("%d ", temp);
    }
}

int main() {
    int a[100];
    int j = 0, i;

    while (scanf("%d", &i) && i != 0) {
        a[j++] = i;
    }

    calculate(a, j);

    return 0;
}
```

## 4. 点击消除

**分值：** 10 分

**题目描述**  
给定一个字符串，每次"点击"，可以把字符串中相邻两个相同字母消除。例如，字符串 `abbc` 点击后可以生成 `ac`。但相同而不相邻、不相同的相邻字母都是不可以被消除的。  
通过点击足够多次之后可以把字符串变得尽可能短，编程实现输出给定字符串足够多次点击后的最终形态。

**输入描述**  
一个字符串，仅由小写字母组成（字符串长度不大于 10000）。

**输出描述**  
一个字符串，为"点击消除"后的最终形态。若最终的字符串为空串，则输出 `0`。

**输入输出样例**

**输入**

```
abba
```

**输出**

```
0
```

```c
#include <stdio.h>
#include <string.h>

#define MAXSIZE 10001

void deletesame(char* a) {
    int i = 0;

    while (a[i] != '\0') {
        // 发现相邻相同
        if (i >= 0 && a[i] == a[i + 1] && a[i] != '\0') {
            // 将后面所有的字符向前平移 2 位
            int j = i;
            while (a[j + 2] != '\0') {
                a[j] = a[j + 2];
                j++;
            }
            a[j] = '\0';

            // 消除后位置回退，检查新挪过来的字符是否和前一个组成一对
            if (i > 0) i--;
            else i = 0;
        } else {
            i++;
        }
    }
}

int main() {
    char a[MAXSIZE];
    if (scanf("%s", a) == EOF) return 0;

    deletesame(a);

    if (a[0] == '\0') {
        printf("0\n");
    } else {
        printf("%s\n", a);
    }

    return 0;
}
```

## 6. 无法吃午餐的学生数量

**分值：** 10 分

**题目描述**  
学校的自助午餐提供圆形和方形的三明治，分别用数字 `0` 和 `1` 表示。所有学生站在一个队列里，每个学生要么喜欢圆形的要么喜欢方形的。餐厅里三明治的数量与学生的数量相同。所有三明治都放在一个栈里，每一轮：

1. 如果队列最前面的学生喜欢栈顶的三明治，那么会拿走它并离开队列。
2. 否则，这名学生会放弃这个三明治并回到队列的尾部。
3. 这个过程会一直持续到队列里所有学生都不喜欢栈顶的三明治为止。

给你两个整数数组 `students` 和 `sandwiches`，其中 `sandwiches[i]` 是栈里面第 `i` 个三明治的类型，`students[j]` 是初始队列里第 `j` 名学生对三明治的喜好。请你返回无法吃午餐的学生数量。

**输入描述**  
输入为两行：  
第一行数字表示学生对三明治的喜好（`-1` 表示结束）。  
第二行数字表示三明治的顺序（`-1` 表示结束）。  
数组中元素数量均 < 100。

**输出描述**  
输出为一个整数，表示无法吃午餐的学生的数量。

**输入输出样例**

**输入**

```
1 1 0 0 -1
0 1 0 1 -1
```

**输出**

```
0
```

```c
#include <stdio.h>

int main() {
    int students[101];
    int sandwiches[101];
    int count0 = 0;  // 喜欢圆形(0)的学生数量
    int count1 = 0;  // 喜欢方形(1)的学生数量
    int s_count = 0; // 三明治总数
    int val;

    // 1. 读取学生喜好，并统计两类人的数量
    while (scanf("%d", &val) && val != -1) {
        if (val == 0) count0++;
        else if (val == 1) count1++;
    }

    // 2. 读取三明治顺序
    while (scanf("%d", &val) && val != -1) {
        sandwiches[s_count++] = val;
    }

    // 3. 遍历三明治栈
    int i;
    for (i = 0; i < s_count; i++) {
        if (sandwiches[i] == 0) {
            if (count0 > 0) count0--;
            else break;
        } else {
            if (count1 > 0) count1--;
            else break;
        }
    }

    // 4. 剩余学生数 = 总数 - 已吃掉的数量
    printf("%d\n", s_count - i);

    return 0;
}
```

## 7. 栈的压入、弹出序列

**分值：** 10 分

**题目描述**  
对输入的两个整数序列，第一个序列表示栈的压入顺序，请判断第二个序列是否可能为该栈的弹出顺序。假设压入栈的所有数字均不相等。例如序列 `1,2,3,4,5` 是某栈的压入顺序，序列 `4,5,3,2,1` 是该压栈序列对应的一个弹出序列，但 `4,3,5,1,2` 就不可能是该压栈序列的弹出序列。

**输入描述**  
输入为两行，每行包括一个数字序列，用空格隔开，`-1` 结束。两个数字序列分别表示栈的压入顺序与弹出顺序。栈中数据的数量 ≤ 1000。

**输出描述**  
输出为 `true` / `false`，表示第二个序列能否为该栈的弹出顺序。

**输入输出样例**

**输入**

```
1 2 3 4 5 -1
4 5 3 2 1 -1
```

**输出**

```
true
```

```c
#include <stdio.h>

#define MAXSIZE 100

typedef struct {
    int data[MAXSIZE];
    int top;
} Stack;

void initStack(Stack *s) {
    s->top = -1;
}

void push(Stack *s, int a) {
    if (s->top == MAXSIZE - 1) return;
    s->data[++s->top] = a;
}

void pop(Stack *s) {
    if (s->top == -1) return;
    s->top--;
}

int judgement(int *a, int n, int *b) {
    Stack s;
    initStack(&s);

    int j = 0;

    for (int i = 0; i < n; i++) {
        push(&s, a[i]);

        while (1) {
            if (s.top == -1) break;
            if (s.data[s.top] == b[j]) {
                pop(&s);
                j++;
            } else break;
        }
    }

    return s.top == -1 ? 1 : 0;
}

int main() {
    int a[MAXSIZE], b[MAXSIZE];
    int n = 0, m = 0;
    int i;

    while (scanf("%d", &i) && i != -1) {
        a[n++] = i;
    }

    while (scanf("%d", &i) && i != -1) {
        b[m++] = i;
    }

    int t = judgement(a, n, b);

    if (t == 0) printf("false");
    else printf("true");

    return 0;
}
```

## 8. 循环队列

**分值：** 10 分

**题目描述**  
请你实现一个循环队列，该循环队列可利用的空间大小等于 `n` 个 `int` 型变量的大小。  
操作：

- `push x`：将 `x` 加入到循环队列尾端。若循环队列已满，输出 `full`。
- `front`：输出队首元素，队首不出队。若队列为空，输出 `empty`。
- `pop`：输出队首元素，且队首出队。若队列为空，输出 `empty`。

**输入描述**  
第一行输入两个整数 `n, q`（`1 ≤ n, q ≤ 10^5`），表示循环队列可利用的空间大小和操作次数。  
接下来的 `q` 行，每行一个操作命令。

**输出描述**  
按照题目描述中的要求输出。

**输入输出样例**

**输入**

```
3 5
push 1
push 2
front
pop
push 3
```

**输出**

```
1
1
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int *data;
    int front;
    int rear;
    int size;
    int capacity;
} Queue;

void initQueue(Queue *q, int n) {
    q->data = (int*)malloc(sizeof(int) * n);
    q->front = 0;
    q->rear = -1;
    q->size = 0;
    q->capacity = n;
}

void push(Queue *q, int x) {
    if (q->size == q->capacity) {
        printf("full\n");
        return;
    }
    q->rear = (q->rear + 1) % q->capacity;
    q->data[q->rear] = x;
    q->size++;
}

void front(Queue *q) {
    if (q->size == 0) {
        printf("empty\n");
        return;
    }
    printf("%d\n", q->data[q->front]);
}

void pop(Queue *q) {
    if (q->size == 0) {
        printf("empty\n");
        return;
    }
    printf("%d\n", q->data[q->front]);
    q->front = (q->front + 1) % q->capacity;
    q->size--;
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);

    Queue q;
    initQueue(&q, n);

    char temp[10];
    int x;

    for (int i = 0; i < k; i++) {
        scanf("%s", temp);

        if (strcmp(temp, "push") == 0) {
            scanf("%d", &x);
            push(&q, x);
        } else if (strcmp(temp, "front") == 0) {
            front(&q);
        } else {
            pop(&q);
        }
    }

    return 0;
}
```

## 9. 最长有效括号

**分值：** 10 分

**题目描述**  
给你一个只包含 `(` 和 `)` 的字符串，找出最长有效（格式正确且连续）括号子串的长度。

**输入描述**  
输入为一个只包含 `(` 和 `)` 的字符串，字符串长度 < 10000。

**输出描述**  
输出为一个数字，表示最长有效括号子串的长度。

**输入输出样例**

**输入**

```
(()()
```

**输出**

```
2
```

```c
#include <stdio.h>
#include <string.h>

#define MAXSIZE 10005

// 定义栈结构，用来存储括号的"下标"
typedef struct {
    int data[MAXSIZE];
    int top;
} Stack;

void initstack(Stack* s) {
    s->top = -1;
}

void push(Stack* s, int val) {
    if (s->top < MAXSIZE - 1) {
        s->data[++(s->top)] = val;
    }
}

void pop(Stack* s) {
    if (s->top != -1) {
        s->top--;
    }
}

// 获取最长有效括号长度
int validlength(char* a) {
    Stack s;
    initstack(&s);

    int maxLen = 0;

    // 初始压入 -1 作为基准参照点
    push(&s, -1);

    for (int i = 0; a[i] != '\0'; i++) {
        if (a[i] == '(') {
            // 遇到左括号，把它的下标压入栈
            push(&s, i);
        } else {
            // 遇到右括号，先弹出一个元素
            pop(&s);

            if (s.top == -1) {
                // 栈空了，说明刚才弹出的是参照点
                // 当前右括号成为新的参照断点
                push(&s, i);
            } else {
                // 匹配成功，计算当前长度
                int currentLen = i - s.data[s.top];
                if (currentLen > maxLen) {
                    maxLen = currentLen;
                }
            }
        }
    }

    return maxLen;
}

int main() {
    char a[MAXSIZE];
    if (scanf("%s", a) == 1) {
        printf("%d", validlength(a));
    }
    return 0;
}
```

## 10. 找出游戏的获胜者

**分值：** 10 分

**题目描述**  
共有 `n` 名小伙伴一起做游戏。小伙伴们围成一圈，按顺时针顺序从 `1` 到 `n` 编号。游戏遵循如下规则：

1. 从第 `1` 名小伙伴所在位置开始。
2. 沿着顺时针方向数 `k` 名小伙伴（计数时包含起始的那位小伙伴）。
3. 你数到的最后一名小伙伴需要离开圈子。
4. 如果圈子中仍然有不止一名小伙伴，从刚刚输掉的小伙伴的顺时针下一位开始，回到步骤 2 继续执行。
5. 否则，圈子中最后一名小伙伴赢得游戏。

给你参与游戏的小伙伴总数 `n` 和一个整数 `k`，返回游戏的获胜者。

**输入描述**  
输入为两个整数 `n` 和 `k`（`1 ≤ k ≤ n ≤ 500`）。

**输出描述**  
输出游戏的获胜者。

**输入输出样例**

**输入**

```
5 2
```

**输出**

```
3
```

```c
#include <stdio.h>

int main() {
    int n, k_step;
    if (scanf("%d %d", &n, &k_step) != 2) return 0;

    int a[501]; // 1 表示在圈子里
    for (int i = 1; i <= n; i++) {
        a[i] = 1;
    }

    int out_count = 0;
    int pos = 1;

    while (out_count < n - 1) {
        int count = 0;

        while (count < k_step) {
            if (a[pos] == 1) count++;
            if (count < k_step) {
                pos++;
                if (pos > n) pos = 1;
            }
        }

        // 淘汰第 k 个
        a[pos] = 0;
        out_count++;

        // 找到下一个在圈中的人
        while (a[pos] == 0 && out_count < n - 1) {
            pos++;
            if (pos > n) pos = 1;
        }
    }

    for (int i = 1; i <= n; i++) {
        if (a[i] == 1) {
            printf("%d\n", i);
            break;
        }
    }

    return 0;
}
```
