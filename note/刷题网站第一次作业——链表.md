## 1. 合并两个有序链表

**分值：** 10 分

**题目描述**  
将两个升序链表合并为一个新的升序链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。

**输入描述**  
输入两个升序链表 `l1`、`l2`，两个链表的节点数目范围是 `[0, 50]`，`-100 <= Node.val <= 100`，`-1` 表示链表结束符。

**输出描述**  
输出合并后的升序链表。

**输入输出样例**

**输入**

```
1 2 4 -1
1 3 4 -1
```

**输出**

```
1 1 2 3 4 4
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* createList() {
    Node* head = NULL;
    Node* trail = NULL;
    int i;

    while (scanf("%d", &i) && i != -1) {
        Node* newnode = (Node*)malloc(sizeof(Node));
        newnode->data = i;
        newnode->next = NULL;

        if (head == NULL) {
            head = trail = newnode;
        } else {
            trail->next = newnode;
            trail = newnode;
        }
    }
    return head;
}

Node* merge(Node* L1, Node* L2) {
    Node L3;
    L3.next = NULL;
    Node* p = &L3;

    while (L1 && L2) {
        if (L1->data <= L2->data) {
            p->next = L1;
            L1 = L1->next;
        } else {
            p->next = L2;
            L2 = L2->next;
        }
        p = p->next;
    }

    p->next = L1 ? L1 : L2;
    return L3.next;
}

int main() {
    Node* L1 = createList();
    Node* L2 = createList();
    Node* L3 = merge(L1, L2);

    Node* p = L3;
    while (p) {
        printf("%d ", p->data);
        p = p->next;
    }

    return 0;
}
```

## 2. 删除排序链表中的重复元素

**分值：** 10 分

**题目描述**  
给定一个已排序的链表的头 `head`，删除所有重复的元素，使每个元素只出现一次。返回已排序的链表。

**输入描述**  
输入已排序的链表，链表中节点数目在范围 `[0, 300]` 内，`-100 <= Node.val <= 100`，题目数据保证链表已经按升序排列，`-1` 表示链表结束符。

**输出描述**  
输出删除重复元素后的链表。

**输入输出样例**

**输入**

```
1 1 2 3 3 -1
```

**输出**

```
1 2 3
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* creatList() {
    Node* head = NULL;
    Node* trail = NULL;
    int i;

    while (scanf("%d", &i) && i != -1) {
        Node* newnode = (Node*)malloc(sizeof(Node));
        newnode->data = i;
        newnode->next = NULL;

        if (head == NULL) {
            head = trail = newnode;
        } else {
            trail->next = newnode;
            trail = newnode;
        }
    }
    return head;
}

// 删除重复节点（保留第一个）这里的删除是假设链表没排序的，适用情况更多
void deletenode(Node* head) {
    Node* p1 = head;

    while (p1) {
        Node* p2 = p1;  // 从 p1 开始
        while (p2->next) {
            if (p2->next->data == p1->data) {
                p2->next = p2->next->next;  // 删除
            } else {
                p2 = p2->next;
            }
        }
        p1 = p1->next;
    }
}

int main() {
    Node* L1 = creatList();
    deletenode(L1);

    while (L1) {
        printf("%d ", L1->data);
        L1 = L1->next;
    }

    return 0;
}
```

## 3. 移除链表元素

给你一个链表的头节点 `head` 和一个整数 `val`，请你删除链表中所有满足 `Node.val == val` 的节点，并返回新的头节点。

**输入描述**  
输入链表和 `val`，列表中的节点数目在范围 `[0, 10^4]` 内，`1 <= Node.val <= 50`，`0 <= val <= 50`，`-1` 表示链表结束符。

**输出描述**  
输出移除元素后的链表。

**输入输出样例**

**输入**

```
1 2 6 3 4 5 6 -1 6
```

**输出**

```
1 2 3 4 5
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* createList() {
    Node* head = NULL;

    int i = 1;

    while (scanf("%d", &i) && i != -1) {
        Node* newnode = (Node*)malloc(sizeof(Node));
        newnode->data = i;
        newnode->next = NULL;

        if (head == NULL) {
            head = trail = newnode;
        } else {
            trail->next = newnode;
            trail = newnode;
        }
    }
    return head;
}

    Node* dummy = (Node*)malloc(sizeof(Node));
    dummy->next = head;
    Node* p = dummy;

    while (p->next) {
        if (p->next->data == value) {
            p->next = p->next->next;
        } else {
            p = p->next;
        }
    }
    return dummy->next;
}

void printfList(Node* head) {
    while (head) {
        printf("%d", head->data);
        if (head->next) printf(" ");
        head = head->next;
    }
}

int main() {
    int value;
    Node* L1 = createList();
    scanf("%d", &value);

    L1 = deleteelem(L1, value);
    printfList(L1);

    return 0;
}
```

## 5. 合并两个链表

**分值：** 10 分

**题目描述**  
给你两个链表 `list1` 和 `list2`，它们包含的元素分别为 `n` 个和 `m` 个。请你将 `list1` 中下标从 `a` 到 `b` 的全部节点都删除，并将 `list2` 接在被删除节点的位置。

**输入描述**  
`3 <= list1.length <= 10^4`，`1 <= a <= b < list1.length - 1`，`1 <= list2.length <= 10^4`，`-1` 作为链表结束符。

**输出描述**  
输出合并后的链表。

**输入输出样例**

**输入**

```
0 1 2 3 4 5 -1
3
4
1000000 1000001 1000002 -1
```

**输出**

```
0 1 2 1000000 1000001 1000002 5
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

// 创建链表（尾插法）
Node* createL() {
    Node* head = NULL;
    Node* tail = NULL;
    int x;

    while (scanf("%d", &x) && x != -1) {
        Node* node = (Node*)malloc(sizeof(Node));
        node->data = x;
        node->next = NULL;

        if (head == NULL) {
            head = tail = node;
        } else {
            tail->next = node;
            tail = node;
        }
    }
    return head;
}

// 核心函数：删除 [a,b] 并接入 L2
Node* mergeL(Node* L1, Node* L2, int a, int b) {
    Node dummy;
    dummy.next = L1;

    Node* p1 = &dummy;
    int i = 0;

    // 找到 a-1 位置
    while (i < a) {
        p1 = p1->next;
        i++;
    }

    Node* p2 = p1;

    // 找到 b 位置
    i = a;
    while (i <= b) {
        p2 = p2->next;
        i++;
    }

    // 找 L2 尾节点
    Node* tail = L2;
    while (tail->next) {
        tail = tail->next;
    }

    // 拼接
    p1->next = L2;
    tail->next = p2->next;

    return dummy.next;
}

// 输出链表
void printList(Node* head) {
    while (head) {
        printf("%d", head->data);
        if (head->next) printf(" ");
        head = head->next;
    }
}

int main() {
    Node* L1 = createL();

    int a, b;
    scanf("%d %d", &a, &b);

    Node* L2 = createL();

    L1 = mergeL(L1, L2, a, b);
    printList(L1);

    return 0;
}
```

## 6. 二进制链表转整数

**分值：** 10 分

**题目描述**  
给你一个单链表的引用结点 `head`。链表中每个结点的值不是 `0` 就是 `1`。已知此链表是一个整数数字的二进制表示形式。请你返回该链表所表示数字的十进制值。

**输入描述**  
输入链表，链表不为空。链表的结点总数不超过 `30`。每个结点的值不是 `0` 就是 `1`。`-1` 表示链表结束符。

**输出描述**  
输出十进制数。

**输入输出样例**

**输入**

```
1 0 1 -1
```

**输出**

```
5
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* createL() {
    Node* head = (Node*)malloc(sizeof(Node));
    head->next = NULL;
    int i = 0;

    while (scanf("%d", &i) && i != -1) {
        Node* newNode = (Node*)malloc(sizeof(Node));
        newNode->data = i;
        newNode->next = NULL;

        newNode->next = head->next;
        head->next = newNode;
    }
    return head->next;
}

int exponential(int i) {
    int t = 1;
    for (int n = 0; n < i; n++) {
        t *= 2;
    }
    return t;
}

int transform(Node* head) {
    int i = 0, t = 0;
    Node* p = head;

    while (p) {
        t += p->data * exponential(i);
        p = p->next;
        i++;
    }
    return t;
}

int main() {
    Node* L = createL();
    printf("%d", transform(L));
    return 0;
}
```

## 7. 两数相加 II

**分值：** 10 分

**题目描述**  
给你两个非空链表来代表两个非负整数。数字最高位位于链表开始位置。它们的每个节点只存储一位数字。将这两数相加会返回一个新的链表。你可以假设除了数字 `0` 之外，这两个数字都不会以零开头。

**输入描述**  
链表的长度范围为 `[1, 100]`，`0 <= node.val <= 9`，输入数据保证链表代表的数字无前导 `0`，`-1` 表示链表结束符。

**输出描述**  
输出相加后的链表。

**输入输出样例**

**输入**

```
7 2 4 3 -1
5 6 4 -1
```

**输出**

```
7 8 0 7
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node
{
    int val;
    struct Node* next;
}Node;

Node* createList()
{
    Node *head=NULL,*tail=NULL;
    int x;

    while(scanf("%d",&x) && x!=-1)
    {
        Node* node=(Node*)malloc(sizeof(Node));
        node->val=x;
        node->next=NULL;

        if(head==NULL)
        {
            head=tail=node;
        }
        else
        {
            tail->next=node;
            tail=node;
        }
    }

    return head;
}

Node* addTwoNumbers(Node* l1,Node* l2)
{
    int s1[105],s2[105];
    int top1=0,top2=0;

    Node* p=l1;
    while(p)
    {
        s1[top1++]=p->val;
        p=p->next;
    }

    p=l2;
    while(p)
    {
        s2[top2++]=p->val;
        p=p->next;
    }

    int carry=0;

    Node* head=NULL;

    while(top1||top2||carry)
    {
        int sum=carry;

        if(top1)
            sum+=s1[--top1];

        if(top2)
            sum+=s2[--top2];

        carry=sum/10;

        Node* node=(Node*)malloc(sizeof(Node));
        node->val=sum%10;

        node->next=head;
        head=node;
    }

    return head;
}

void printList(Node* head)
{
    while(head)
    {
        printf("%d",head->val);

        if(head->next)
            printf(" ");

        head=head->next;
    }
}

int main()
{
    Node* l1=createList();
    Node* l2=createList();

    Node* ans=addTwoNumbers(l1,l2);

    printList(ans);

    return 0;
}
```

## 8. 反转链表 II

**分值：** 10 分

**题目描述**  
给你单链表的头指针 `head` 和两个整数 `left` 和 `right`，其中 `left <= right`。请你反转从位置 `left` 到位置 `right` 的链表节点，返回反转后的链表。

**输入描述**  
链表中节点数目为 `n`，`1 <= n <= 500`，`-500 <= Node.val <= 500`，`1 <= left <= right <= n`，`-1` 表示链表结束符。

**输出描述**  
输出反转后的链表。

**输入输出样例**

**输入**

```
1 2 3 4 5 -1 2 4
```

**输出**

```
1 4 3 2 5
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

// 创建链表（尾插法）
Node* createL() {
    Node* head = NULL;
    Node* tail = NULL;
    int x;

    while (scanf("%d", &x) && x != -1) {
        Node* node = (Node*)malloc(sizeof(Node));
        node->data = x;
        node->next = NULL;

        if (head == NULL) {
            head = tail = node;
        } else {
            tail->next = node;
            tail = node;
        }
    }
    return head;
}

// 反转区间 [a, b]
Node* inverseL(Node* L, int a, int b) {
    Node dummy;
    dummy.next = L;

    Node* p1 = &dummy;

    // 找到 a-1 位置
    for (int i = 1; i < a; i++) {
        p1 = p1->next;
    }

    // 从 a 开始反转
    Node* cur = p1->next;  // a
    Node* pre = NULL;
    Node* nex = NULL;

    // 反转 b-a+1 个节点
    for (int i = 0; i <= b - a; i++) {
        nex = cur->next;
        cur->next = pre;
        pre = cur;
        cur = nex;
    }

    // 拼接
    Node* p2 = p1->next;  // 原来的 a（现在是尾）
    p1->next = pre;       // 接上反转后的头
    p2->next = cur;       // 接上后半部分

    return dummy.next;
}

// 输出链表
void printList(Node* head) {
    while (head) {
        printf("%d", head->data);
        if (head->next) printf(" ");
        head = head->next;
    }
}

int main() {
    Node* L = createL();

    int a, b;
    scanf("%d %d", &a, &b);

    L = inverseL(L, a, b);

    printList(L);

    return 0;
}
```

## 9. 链表最大孪生和

**分值：** 10 分

**题目描述**  
在一个大小为 `n` 且 `n` 为偶数的链表中，对于 `0 <= i <= (n / 2) - 1` 的 `i`，第 `i` 个节点（下标从 `0` 开始）的孪生节点为第 `(n-1-i)` 个节点。孪生和定义为一个节点和它孪生节点两者值之和。  
给你一个长度为偶数的链表的头节点 `head`，请你返回链表的最大孪生和。

**输入描述**  
输入链表的节点数目是 `[2, 10^5]` 中的偶数，`1 <= Node.val <= 10^5`，`-1` 是链表结束符。

**输出描述**  
输出最大孪生和。

**输入输出样例**

**输入**

```
5 4 2 1 -1
```

**输出**

```
6
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* createL() {
    Node* head = NULL;
    Node* trail = NULL;
    int i = 0;

    while (scanf("%d", &i) && i != -1) {
        Node* newNode = (Node*)malloc(sizeof(Node));
        newNode->next = NULL;
        newNode->data = i;

        if (head == NULL)
            head = trail = newNode;
        else {
            trail->next = newNode;
            trail = newNode;
        }

}

Node* copy(Node* L) {
    Node* head = NULL;
    Node* trail = NULL;

    while (L) {
        Node* newNode = (Node*)malloc(sizeof(Node));
        newNode->data = L->data;
        newNode->next = NULL;

        if (head == NULL)
            head = trail = newNode;
        else {
            trail->next = newNode;
            trail = newNode;
        }
        L = L->next;
    }
    return head;
}

    Node* pre = NULL;
    Node* cur = L;
    Node* nex = NULL;

    while (cur) {
        nex = cur->next;
        cur->next = pre;
        pre = cur;
        cur = nex;
    }
    return pre;
}

int main() {

    Node* L3 = copy(L1);
    Node* L2 = inverse(L3);

    int n = 0;
    Node* p = L1;

        p = p->next;

    }
```

## 10. 重排链表

**分值：** 10 分

**题目描述**  
给定一个单链表 `L` 的头节点 `head`，单链表 `L` 表示为：`L0 → L1 → … → Ln-1 → Ln`。  
请将其重新排列后变为：`L0 → Ln → L1 → Ln-1 → L2 → Ln-2 → …`。  
不能只是单纯的改变节点内部的值，而是需要实际进行节点交换。

**输入描述**  
输入链表的长度范围为 `[1, 5 * 10^4]`，`1 <= node.val <= 1000`，`-1` 表示链表结束符。

**输出描述**  
输出重排后的链表。

**输入输出样例**

**输入**

```
1 2 3 4 5 -1
```

**输出**

```
1 5 2 4 3
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* createL() {
    Node* head = NULL;
    Node* trail = NULL;
    int i = 0;

    while (scanf("%d", &i) && i != -1) {
        Node* newNode = (Node*)malloc(sizeof(Node));
        newNode->data = i;
        newNode->next = NULL;

        if (head == NULL)
            head = trail = newNode;
        else {
            trail->next = newNode;
            trail = newNode;
        }
    }
    return head;
}

Node* inverse(Node* L) {
    Node* pre = NULL;
    Node* cur = L;
    Node* nex = NULL;

    while (cur) {
        nex = cur->next;
        cur->next = pre;
        pre = cur;
        cur = nex;
    }
    return pre;
}

Node* resort(Node* L1, Node* L2) {
    Node* head = L1;
    Node* p1 = L1;
    Node* p2 = L1;
    Node* p3 = L2;
    Node* p4 = L2;

    while (p3 && p1) {
        p1 = p1->next;
        p3 = p3->next;
        p4->next = p2->next;
        p2->next = p4;
        p2 = p1;
        p4 = p3;
    }
    return head;
}

int main() {
    Node* L1 = createL();

    Node* slow = L1;
    Node* fast = L1;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
    }

    Node* L2 = inverse(slow->next);
    slow->next = NULL;

    Node* L3 = resort(L1, L2);

    while (L3) {
        printf("%d", L3->data);
        if (L3->next) printf(" ");
        L3 = L3->next;
    }

    return 0;
}
```

## 11. 删除链表的倒数第 N 个结点

**分值：** 10 分

**题目描述**  
给你一个链表，删除链表的倒数第 `n` 个结点，并且返回链表的头结点。

**输入描述**  
输入链表和要删除的 `n`，链表中结点的数目为 `sz`，`1 <= sz <= 30`，`0 <= Node.val <= 100`，`1 <= n <= sz`，`-1` 表示链表结束符。

**输出描述**  
输出删除后的链表。

**输入输出样例**

**输入**

```
1 2 3 4 5 -1
2
```

**输出**

```
1 2 3 5
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

// 创建链表（尾插法）
Node* createL() {
    Node *head = NULL, *tail = NULL;
    int x;

    while (scanf("%d", &x) && x != -1) {
        Node* newNode = (Node*)malloc(sizeof(Node));
        newNode->data = x;
        newNode->next = NULL;

        if (head == NULL) {
            head = tail = newNode;
        } else {
            tail->next = newNode;
            tail = newNode;
        }
    }
    return head;
}

// 删除倒数第 n 个节点
Node* removeNthFromEnd(Node* head, int n) {
    Node* dummy = (Node*)malloc(sizeof(Node));
    dummy->next = head;

    Node *fast = dummy, *slow = dummy;

    // fast 先走 n 步
    for (int i = 0; i < n; i++) {
        fast = fast->next;
    }

    // 一起走
    while (fast->next != NULL) {
        fast = fast->next;
        slow = slow->next;
    }

    // 删除节点
    Node* temp = slow->next;
    slow->next = temp->next;
    free(temp);

    return dummy->next;
}

// 打印链表
void printL(Node* head) {
    while (head) {
        printf("%d ", head->data);
        head = head->next;
    }
}

int main() {
    Node* head = createL();
    int n;
    scanf("%d", &n);

    head = removeNthFromEnd(head, n);
    printL(head);

    return 0;
}
```
