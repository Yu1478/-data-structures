## 1. 相同的树

**分值：** 10 分

**题目描述**  
给你两棵二叉树的根节点 `p` 和 `q`，编写一个函数来检验这两棵树是否相同。  
如果两个树在结构上相同，并且节点具有相同的值，则认为它们是相同的。

**输入输出样例**

**输入**

```
p = 1,2,3, q = 1,2,3
```

**输出**

```
true
```

**输入**

```
p = 1,2, q = 1,null,2
```

**输出**

```
false
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
} TreeNode;

// 创建节点
TreeNode* newNode(int x) {
    TreeNode* node = (TreeNode*)malloc(sizeof(TreeNode));
    node->val = x;
    node->left = NULL;
    node->right = NULL;
    return node;
}

// 层序建树
TreeNode* buildTree(char s[]) {
    // 空树
    if (strcmp(s, "[]") == 0)
        return NULL;

    // 去掉 '[' 和 ']'
    int len = strlen(s);
    s[len - 1] = '\0';
    char *p = s + 1;

    // 分割字符串
    char *token = strtok(p, ",");
    if (token == NULL)
        return NULL;

    // 根节点
    TreeNode* root = newNode(atoi(token));

    // 队列
    TreeNode* queue[1000];
    int front = 0, rear = 0;
    queue[rear++] = root;

    while (front < rear) {
        TreeNode* cur = queue[front++];

        // 左孩子
        token = strtok(NULL, ",");
        if (token != NULL) {
            if (strcmp(token, "null") != 0) {
                cur->left = newNode(atoi(token));
                queue[rear++] = cur->left;
            }
        }

        // 右孩子
        token = strtok(NULL, ",");
        if (token != NULL) {
            if (strcmp(token, "null") != 0) {
                cur->right = newNode(atoi(token));
                queue[rear++] = cur->right;
            }
        }
    }

    return root;
}

// 判断是否相同
int isSameTree(TreeNode* t1, TreeNode* t2) {
    if (t1 == NULL && t2 == NULL)
        return 1;
    if (t1 == NULL || t2 == NULL)
        return 0;
    if (t1->val != t2->val)
        return 0;
    return isSameTree(t1->left, t2->left) &&
           isSameTree(t1->right, t2->right);
}

int main() {
    char s1[1000], s2[1000];
    scanf("%s", s1);
    scanf("%s", s2);

    TreeNode* t1 = buildTree(s1);
    TreeNode* t2 = buildTree(s2);

    if (isSameTree(t1, t2))
        printf("true");
    else
        printf("false");

    return 0;
}
```

## 2. 对称二叉树

**分值：** 10 分

**题目描述**  
给你一个二叉树的根节点 `root`，检查它是否轴对称。

**输入描述**  
一个节点序列

**输出描述**  
返回 `true` / `false`

**输入输出样例**

**输入**

```
root = 1,2,2,3,4,4,3
```

**输出**

```
true
```

**输入**

```
root = 1,2,2,null,3,null,3
```

**输出**

```
false
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
} TreeNode;

TreeNode* createNode(int val) {
    TreeNode* node = (TreeNode*)malloc(sizeof(TreeNode));
    node->val = val;
    node->left = NULL;
    node->right = NULL;
    return node;
}

/* 层序建树 */
TreeNode* buildTree(char *str) {
    char *tokens[2000];
    int count = 0;
    char *p = strtok(str, ",");

    while (p) {
        while (*p == ' ') p++;
        tokens[count++] = p;
        p = strtok(NULL, ",");
    }

    if (count == 0) return NULL;
    if (strcmp(tokens[0], "null") == 0) return NULL;

    TreeNode *root = createNode(atoi(tokens[0]));
    TreeNode *queue[2000];
    int front = 0, rear = 0;
    queue[rear++] = root;
    int i = 1;

    while (front < rear && i < count) {
        TreeNode *cur = queue[front++];

        if (i < count && strcmp(tokens[i], "null") != 0) {
            cur->left = createNode(atoi(tokens[i]));
            queue[rear++] = cur->left;
        }
        i++;

        if (i < count && strcmp(tokens[i], "null") != 0) {
            cur->right = createNode(atoi(tokens[i]));
            queue[rear++] = cur->right;
        }
        i++;
    }

    return root;
}

/* 判断镜像 */
int isMirror(TreeNode *a, TreeNode *b) {
    if (a == NULL && b == NULL) return 1;
    if (a == NULL || b == NULL) return 0;
    if (a->val != b->val) return 0;
    return isMirror(a->left, b->right) &&
           isMirror(a->right, b->left);
}

int isSymmetric(TreeNode *root) {
    if (root == NULL) return 1;
    return isMirror(root->left, root->right);
}

int main() {
    char line[10000];
    fgets(line, sizeof(line), stdin);
    line[strcspn(line, "\n")] = '\0';

    TreeNode *root = buildTree(line);

    if (isSymmetric(root))
        printf("true");
    else
        printf("false");

    return 0;
}
```

## 3. 翻转二叉树

**分值：** 10 分

**题目描述**  
给你一棵二叉树的根节点 `root`，翻转这棵二叉树，并返回其根节点。翻转二叉树即交换每个节点的左右子树。

**输入描述**  
输入包含一行，为一个以逗号分隔的节点序列，表示二叉树的层序遍历结果。空节点用 `null` 表示。

**输出描述**  
输出翻转后二叉树的层序遍历结果序列，要求使用逗号分隔，并**省略序列末尾的所有 `null`**。

**输入输出样例**

**样例 1**

**输入**

```
4,2,7,1,3,6,9
```

**输出**

```
4,7,2,9,6,3,1
```

**样例 2**

**输入**

```
2,1,3
```

**输出**

```
2,3,1
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct Treenode {
    int val;
    struct Treenode* left;
    struct Treenode* right;
} Treenode;

// 创建节点
Treenode* createnode(int x) {
    Treenode* node = (Treenode*)malloc(sizeof(Treenode));
    node->val = x;
    node->left = NULL;
    node->right = NULL;
    return node;
}

// 构建二叉树
Treenode* buildTree(char* arr[], int size) {
    if (size == 0) return NULL;
    if (strcmp(arr[0], "null") == 0) return NULL;

    Treenode* root = createnode(atoi(arr[0]));
    Treenode* queue[1000];
    int front = 0, rear = 0;
    queue[rear++] = root;
    int i = 1;

    while (front < rear && i < size) {
        Treenode* cur = queue[front++];

        // 左孩子
        if (i < size && strcmp(arr[i], "null") != 0) {
            cur->left = createnode(atoi(arr[i]));
            queue[rear++] = cur->left;
        }
        i++;

        // 右孩子
        if (i < size && strcmp(arr[i], "null") != 0) {
            cur->right = createnode(atoi(arr[i]));
            queue[rear++] = cur->right;
        }
        i++;
    }

    return root;
}

// 翻转二叉树
void invert(Treenode* root) {
    if (root == NULL) return;

    Treenode* temp = root->left;
    root->left = root->right;
    root->right = temp;

    invert(root->left);
    invert(root->right);
}

// 输出层序遍历（省略末尾 null）
void printtree(Treenode* root) {
    if (root == NULL) return;

    Treenode* queue[2000];
    int front = 0, rear = 0;
    queue[rear++] = root;

    char ans[2000][20];
    int count = 0;

    while (front < rear) {
        Treenode* cur = queue[front++];

        if (cur == NULL) {
            strcpy(ans[count++], "null");
        } else {
            sprintf(ans[count++], "%d", cur->val);
            queue[rear++] = cur->left;
            queue[rear++] = cur->right;
        }
    }

    // 删除末尾所有 null
    while (count > 0 && strcmp(ans[count - 1], "null") == 0)
        count--;

    // 输出
    for (int i = 0; i < count; i++) {
        if (i != 0) printf(",");
        printf("%s", ans[i]);
    }
}

int main() {
    char s[1000];
    scanf("%s", s);

    // 拆分字符串
    char* arr[1000];
    int size = 0;
    char* token = strtok(s, ",");

    while (token != NULL) {
        arr[size++] = token;
        token = strtok(NULL, ",");
    }

    Treenode* root = buildTree(arr, size);
    invert(root);
    printtree(root);

    return 0;
}
```

## 4. 左叶子之和

**分值：** 10 分

**题目描述**  
给定二叉树的根节点 `root`，返回所有左叶子之和。

**输入描述**  
一个节点序列

**输出描述**  
返回其左叶子之和

**输入输出样例**

**输入**

```
root = 3,9,20,null,null,15,7
```

**输出**

```
24
```

**输入**

```
root = 1
```

**输出**

```
0
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

typedef struct TreeNode {
    int val;
    struct TreeNode* left;
    struct TreeNode* right;
} TreeNode;

TreeNode* createNode(int x) {
    TreeNode* node = (TreeNode*)malloc(sizeof(TreeNode));
    node->val = x;
    node->left = NULL;
    node->right = NULL;
    return node;
}

// 建树（从带 root = 前缀或纯序列的字符串解析）
TreeNode* buildTree(char str[]) {
    char tokens[10000][20];
    int count = 0;
    int i = 0;

    while (str[i]) {
        // 跳过无关字符
        while (str[i] && (str[i] == ' ' || str[i] == '[' ||
               str[i] == ']' || str[i] == '=' ||
               str[i] == ',' || str[i] == '\n' || str[i] == '\r'))
            i++;

        if (str[i] == '\0') break;

        int j = 0;
        while (str[i] && str[i] != ' ' && str[i] != ',' &&
               str[i] != ']' && str[i] != '\n' && str[i] != '\r') {
            // 跳过 "root"
            if (strncmp(&str[i], "root", 4) == 0) {
                i += 4;
                break;
            }
            tokens[count][j++] = str[i++];
        }

        if (j > 0) {
            tokens[count][j] = '\0';
            count++;
        }
    }

    if (count == 0) return NULL;
    if (strcmp(tokens[0], "null") == 0) return NULL;

    TreeNode* root = createNode(atoi(tokens[0]));
    TreeNode* queue[10000];
    int front = 0, rear = 0;
    queue[rear++] = root;
    int idx = 1;

    while (front < rear && idx < count) {
        TreeNode* cur = queue[front++];

        if (idx < count && strcmp(tokens[idx], "null") != 0) {
            cur->left = createNode(atoi(tokens[idx]));
            queue[rear++] = cur->left;
        }
        idx++;

        if (idx < count && strcmp(tokens[idx], "null") != 0) {
            cur->right = createNode(atoi(tokens[idx]));
            queue[rear++] = cur->right;
        }
        idx++;
    }

    return root;
}

// 左叶子之和
int sumLeftLeaves(TreeNode* root) {
    if (root == NULL) return 0;

    int sum = 0;
    if (root->left != NULL &&
        root->left->left == NULL &&
        root->left->right == NULL) {
        sum += root->left->val;
    }

    sum += sumLeftLeaves(root->left);
    sum += sumLeftLeaves(root->right);

    return sum;
}

int main() {
    char str[100000];
    fgets(str, sizeof(str), stdin);
    TreeNode* root = buildTree(str);
    printf("%d\n", sumLeftLeaves(root));
    return 0;
}
```

## 5. 最小高度树

**分值：** 10 分

**题目描述**  
给定一个有序整数数组，元素各不相同且按升序排列，编写一个算法，创建一棵高度最小的二叉搜索树。

**输入描述**  
一个有序整数数组

**输出描述**  
高度最小的二叉搜索树（层序遍历，省略末尾 null）

**输入输出样例**

**输入**

```
root = -10,-3,0,5,9
```

**输出**

```
0,-3,9,-10,null,5
```

**输入**

```
root = 1
```

**输出**

```
1
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
} TreeNode;

TreeNode* createNode(int val) {
    TreeNode* node = (TreeNode*)malloc(sizeof(TreeNode));
    node->val = val;
    node->left = NULL;
    node->right = NULL;
    return node;
}

TreeNode* buildBST(int nums[], int left, int right) {
    if (left > right) return NULL;

    // 取右中点（保证高度最小）
    int mid = (left + right + 1) / 2;
    TreeNode* root = createNode(nums[mid]);
    root->left = buildBST(nums, left, mid - 1);
    root->right = buildBST(nums, mid + 1, right);
    return root;
}

void levelOrderPrint(TreeNode* root) {
    if (root == NULL) {
        printf("null");
        return;
    }

    TreeNode* queue[5000];
    int front = 0, rear = 0;
    queue[rear++] = root;

    char ans[5000][20];
    int count = 0;

    while (front < rear) {
        TreeNode* cur = queue[front++];

        if (cur == NULL) {
            strcpy(ans[count++], "null");
        } else {
            sprintf(ans[count++], "%d", cur->val);
            queue[rear++] = cur->left;
            queue[rear++] = cur->right;
        }
    }

    // 去掉末尾 null
    while (count > 0 && strcmp(ans[count - 1], "null") == 0)
        count--;

    for (int i = 0; i < count; i++) {
        if (i) printf(",");
        printf("%s", ans[i]);
    }
}

int main() {
    char str[5000];
    if (fgets(str, sizeof(str), stdin) == NULL)
        return 0;

    int nums[2000];
    int n = 0;
    char *token = strtok(str, ",[] \n\r");

    while (token) {
        nums[n++] = atoi(token);
        token = strtok(NULL, ",[] \n\r");
    }

    TreeNode* root = buildBST(nums, 0, n - 1);
    levelOrderPrint(root);

    return 0;
}
```

## 6. 填充每个节点的下一个右侧节点指针

**分值：** 10 分

**题目描述**  
给定一个 **完美二叉树**，其所有叶子节点都在同一层，每个父节点都有两个子节点。二叉树定义如下：

```c
struct Node {
    int val;
    Node *left;
    Node *right;
    Node *next;
};
```

填充它的每个 `next` 指针，让这个指针指向其下一个右侧节点。如果找不到下一个右侧节点，则将 `next` 指针设置为 `NULL`。  
初始状态下，所有 `next` 指针都被设置为 `NULL`。

**输入描述**  
一个完美二叉树的层序遍历序列

**输出描述**  
按层输出各节点值，每层末尾以 `#` 分隔

**输入输出样例**

**输入**

```
root = 1,2,3,4,5,6,7
```

**输出**

```
1,#,2,3,#,4,5,6,7,#
```

**输入**

```
root = []
```

**输出**

```
[]
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct Node {
    int val;
    struct Node* left;
    struct Node* right;
    struct Node* next;
} Node;

Node* createNode(int x) {
    Node* node = (Node*)malloc(sizeof(Node));
    node->val = x;
    node->left = NULL;
    node->right = NULL;
    node->next = NULL;
    return node;
}

// 层序建树
Node* buildTree(char* arr[], int size) {
    if (size == 0) return NULL;

    Node* root = createNode(atoi(arr[0]));
    Node* queue[1000];
    int front = 0, rear = 0;
    queue[rear++] = root;
    int i = 1;

    while (front < rear && i < size) {
        Node* cur = queue[front++];

        if (i < size && strcmp(arr[i], "null") != 0) {
            cur->left = createNode(atoi(arr[i]));
            queue[rear++] = cur->left;
        }
        i++;

        if (i < size && strcmp(arr[i], "null") != 0) {
            cur->right = createNode(atoi(arr[i]));
            queue[rear++] = cur->right;
        }
        i++;
    }

    return root;
}

// 填充 next 指针（层序遍历）
Node* connect(Node* root) {
    if (root == NULL) return NULL;

    Node* queue[1000];
    int front = 0, rear = 0;
    queue[rear++] = root;

    while (front < rear) {
        int levelSize = rear - front;

        for (int i = 0; i < levelSize; i++) {
            Node* cur = queue[front++];

            if (i < levelSize - 1)
                cur->next = queue[front];
            else
                cur->next = NULL;

            if (cur->left)  queue[rear++] = cur->left;
            if (cur->right) queue[rear++] = cur->right;
        }
    }

    return root;
}

// 按题目要求输出（每层末尾加 #）
void printResult(Node* root) {
    if (root == NULL) return;

    Node* queue[1000];
    int front = 0, rear = 0;
    queue[rear++] = root;
    int first = 1;

    while (front < rear) {
        int levelSize = rear - front;

        for (int i = 0; i < levelSize; i++) {
            Node* cur = queue[front++];

            if (first) {
                printf("%d", cur->val);
                first = 0;
            } else {
                printf(",%d", cur->val);
            }

            if (cur->left)  queue[rear++] = cur->left;
            if (cur->right) queue[rear++] = cur->right;
        }
        printf(",#");
    }
}

int main() {
    char s[5000];
    fgets(s, sizeof(s), stdin);
    s[strcspn(s, "\n")] = '\0';

    char* arr[1000];
    int size = 0;
    char* token = strtok(s, ",");

    while (token != NULL) {
        arr[size++] = token;
        token = strtok(NULL, ",");
    }

    Node* root = buildTree(arr, size);
    root = connect(root);
    printResult(root);

    return 0;
}
```

## 7. 二叉树的最近公共祖先

**分值：** 10 分

**题目描述**  
给定一个二叉树，找到该树中两个指定节点的最近公共祖先。  
（一个节点也可以是它自己的祖先）。

**输入描述**  
第一行：二叉树层序遍历序列  
第二行：`p` 的值  
第三行：`q` 的值

**输出描述**  
输出最近公共祖先的值

**输入输出样例**

**输入**

```
root = 3,5,1,6,2,0,8,null,null,7,4, p = 5, q = 1
```

**输出**

```
3
```

**输入**

```
root = 1,2, p = 1, q = 2
```

**输出**

```
1
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
} TreeNode;

TreeNode* createNode(int val) {
    TreeNode* node = (TreeNode*)malloc(sizeof(TreeNode));
    node->val = val;
    node->left = NULL;
    node->right = NULL;
    return node;
}

/* 查找指定值对应的节点 */
TreeNode* findNode(TreeNode* root, int target) {
    if (root == NULL) return NULL;
    if (root->val == target) return root;

    TreeNode* left = findNode(root->left, target);
    if (left) return left;

    return findNode(root->right, target);
}

/* 最近公共祖先 */
TreeNode* lowestCommonAncestor(TreeNode* root,
                               TreeNode* p,
                               TreeNode* q) {
    if (root == NULL || root == p || root == q)
        return root;

    TreeNode* left = lowestCommonAncestor(root->left, p, q);
    TreeNode* right = lowestCommonAncestor(root->right, p, q);

    if (left && right) return root;
    return left ? left : right;
}

/* 根据层序字符串建树 */
TreeNode* buildTree(char* str) {
    char* tokens[2000];
    int count = 0;
    char* p = strtok(str, ",");

    while (p) {
        while (*p == ' ') p++;
        tokens[count++] = p;
        p = strtok(NULL, ",");
    }

    if (count == 0) return NULL;
    if (strcmp(tokens[0], "null") == 0) return NULL;

    TreeNode* root = createNode(atoi(tokens[0]));
    TreeNode* queue[2000];
    int front = 0, rear = 0;
    queue[rear++] = root;
    int i = 1;

    while (front < rear && i < count) {
        TreeNode* cur = queue[front++];

        if (i < count && strcmp(tokens[i], "null") != 0) {
            cur->left = createNode(atoi(tokens[i]));
            queue[rear++] = cur->left;
        }
        i++;

        if (i < count && strcmp(tokens[i], "null") != 0) {
            cur->right = createNode(atoi(tokens[i]));
            queue[rear++] = cur->right;
        }
        i++;
    }

    return root;
}

int main() {
    char line[10000];
    fgets(line, sizeof(line), stdin);
    line[strcspn(line, "\n")] = '\0';

    int pVal, qVal;
    scanf("%d", &pVal);
    scanf("%d", &qVal);
```

## 8. 恢复二叉搜索树

**题目描述**

**输入描述**  
一个二叉树的层序遍历序列

**输出描述**  
恢复后的二叉树层序遍历序列

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct TreeNode {
    int val;

int idx = 0;

    a[idx++] = root;

    if (root == NULL) return root;

    idx = 0;
    TreeNode* a[1000];
    Inorder(root, a);

                p1 = a[i];
                p2 = a[i + 1];
            } else {
                p2 = a[i + 1];
            }
        }
    }

    if (p1 && p2) {
        int temp = p1->val;
        p1->val = p2->val;
        p2->val = temp;
    }

    return root;
}

        if (pos < cnt) nodes[i]->right = nodes[pos++];

    TreeNode* root = nodes[0];
    restoreBST(root);

    for (int i = 0; i < cnt; i++) {
        if (i) printf(",");
        if (nodes[i] == NULL)
            printf("null");
        else
            printf("%d", nodes[i]->val);
    }

}
```

## 9. 奇偶树

**分值：** 10 分

**题目描述**  
如果一棵二叉树满足下述几个条件，则可以称为 **奇偶树**：

- 二叉树根节点所在层下标为 `0`，根的子节点所在层下标为 `1`，根的孙节点所在层下标为 `2`，依此类推。
- **偶数下标** 层上的所有节点的值都是 **奇数**，从左到右按顺序 **严格递增**。
- **奇数下标** 层上的所有节点的值都是 **偶数**，从左到右按顺序 **严格递减**。

给你二叉树的根节点，如果二叉树为奇偶树，则返回 `true`，否则返回 `false`。

**输入描述**  
一个二叉树的层序遍历序列

**输出描述**  
`true` / `false`

**输入输出样例**

**输入**

```
root = 1,10,4,3,null,7,9,12,8,6,null,null,2
```

**输出**

```
true
```

**输入**

```
root = 2,1,3
```

**输出**

```
false
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
} TreeNode;

/* 创建节点 */
TreeNode* createNode(int val) {
    TreeNode *node = (TreeNode*)malloc(sizeof(TreeNode));
    node->val = val;
    node->left = NULL;
    node->right = NULL;
    return node;
}

/* 队列 */
typedef struct {
    TreeNode *data[5000];
    int front;
    int rear;
} Queue;

void initQueue(Queue *q) {
    q->front = 0;
    q->rear = 0;
}

int empty(Queue *q) {
    return q->front == q->rear;
}

void push(Queue *q, TreeNode *node) {
    q->data[q->rear++] = node;
}

TreeNode* pop(Queue *q) {
    return q->data[q->front++];
}

/* 建树 */
TreeNode* buildTree(char *str) {
    char *tokens[5000];
    int cnt = 0;
    char *token = strtok(str, ",\n\r");

    while (token) {
        tokens[cnt++] = token;
        token = strtok(NULL, ",\n\r");
    }

    if (cnt == 0) return NULL;
    if (strcmp(tokens[0], "null") == 0) return NULL;

    TreeNode *root = createNode(atoi(tokens[0]));
    Queue q;
    initQueue(&q);
    push(&q, root);
    int idx = 1;

    while (!empty(&q) && idx < cnt) {
        TreeNode *cur = pop(&q);

        if (idx < cnt && strcmp(tokens[idx], "null") != 0) {
            cur->left = createNode(atoi(tokens[idx]));
            push(&q, cur->left);
        }
        idx++;

        if (idx < cnt && strcmp(tokens[idx], "null") != 0) {
            cur->right = createNode(atoi(tokens[idx]));
            push(&q, cur->right);
        }
        idx++;
    }

    return root;
}

/* 判断奇偶树 */
int isEvenOddTree(TreeNode *root) {
    if (root == NULL) return 1;

    Queue q;
    initQueue(&q);
    push(&q, root);

    int level = 0;

    while (!empty(&q)) {
        int size = q.rear - q.front;
        int first = 1;
        int prev = 0;

        for (int i = 0; i < size; i++) {
            TreeNode *cur = pop(&q);

            if (level % 2 == 0) {  // 偶数层：奇数且递增
                if (cur->val % 2 == 0) return 0;
                if (!first && cur->val <= prev) return 0;
            } else {  // 奇数层：偶数且递减
                if (cur->val % 2 != 0) return 0;
                if (!first && cur->val >= prev) return 0;
            }

            prev = cur->val;
            first = 0;

            if (cur->left)  push(&q, cur->left);
            if (cur->right) push(&q, cur->right);
        }

        level++;
    }

    return 1;
}

int main() {
    char str[10000];
    fgets(str, sizeof(str), stdin);

    TreeNode *root = buildTree(str);

    if (isEvenOddTree(root))
        printf("true\n");
    else
        printf("false\n");

    return 0;
}
```

## 13. 哈夫曼树的带权路径长度

**分值：** 10 分

**题目描述**  
现在给出一棵二叉树，每个结点有一个权值，请你构造赫夫曼树，并输出其对应的带权路径长度。

**输入描述**  
第一行一个正整数 `n`，表示给定的树的节点的数目。  
第二行 `n` 个正整数，用一个空格分隔，第 `i` 个正整数 `vi`，代表节点 `i` 的权值。

**输出描述**  
一个数，表示对应赫夫曼树的带权路径长度。

**样例**

**输入**

```
3
1 1000 1000
```

**输出**

```
3002
```

```c
#include <stdio.h>

// 用于存放节点权值的数组，建议开大一点并使用 long long 防止溢出
long long arr[10005];

// 在当前有效的节点中寻找两个最小的权值下标
void findTwoMin(int len, int *min1, int *min2) {
    *min1 = -1;
    *min2 = -1;
    for (int i = 0; i < len; i++) {
        if (arr[i] != -1) {  // -1 表示该节点已被合并消耗掉
            if (*min1 == -1 || arr[i] < arr[*min1]) {
                *min2 = *min1;
                *min1 = i;
            } else if (*min2 == -1 || arr[i] < arr[*min2]) {
                *min2 = i;
            }
        }
    }
}

int main() {
    int n;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        scanf("%lld", &arr[i]);
    }

    long long wpl = 0;
    int remain = n;  // 记录当前剩余未合并的节点数量

    while (remain > 1) {
        int m1, m2;
        findTwoMin(n, &m1, &m2);  // 找到当前最小的两个节点

        long long new_weight = arr[m1] + arr[m2];
        wpl += new_weight;  // 累加本次合并产生的路径长度贡献

        // 标记这两个旧节点被消耗
        arr[m1] = -1;
        arr[m2] = -1;

        // 将新生成的父节点放在原 m1 的位置
        arr[m1] = new_weight;
        remain--;  // 每次合并减少一个独立节点
    }

    printf("%lld\n", wpl);
    return 0;
}
```
